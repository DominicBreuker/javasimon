001    package org.javasimon.utils;
002    
003    import org.javasimon.Split;
004    import org.javasimon.StopwatchSample;
005    import org.javasimon.callback.CallbackSkeleton;
006    
007    import java.util.logging.Level;
008    import java.util.logging.Logger;
009    
010    /**
011     * LoggingCallback logs events via JDK logging API.
012     *
013     * @author <a href="mailto:virgo47@gmail.com">Richard "Virgo" Richter</a>
014     */
015    @SuppressWarnings("UnusedDeclaration")
016    public class LoggingCallback extends CallbackSkeleton {
017            private Logger logger = Logger.getAnonymousLogger();
018            private Level level = Level.FINE;
019    
020            /**
021             * Logs Simon start on a sprecified log level.
022             */
023            @Override
024            public void onStopwatchStart(Split split) {
025                    logger.log(level, "SIMON START: " + split.getStopwatch());
026            }
027    
028            /**
029             * Logs Simon stop on a specified log level.
030             */
031            @Override
032            public void onStopwatchStop(Split split, StopwatchSample sample) {
033                    logger.log(level, "SIMON STOP: " + sample.stopwatchToString() + " (" + split.runningFor() + ")");
034            }
035    
036            /**
037             * Logs the warning on a specified log level.
038             *
039             * @param warning warning message
040             * @param cause throwable cause
041             */
042            @Override
043            public void onManagerWarning(String warning, Exception cause) {
044                    logger.log(level, "SIMON WARNING: " + warning, cause);
045            }
046    
047            /**
048             * Logs the message on a specified log level.
049             *
050             * @param message message
051             */
052            @Override
053            public void onManagerMessage(String message) {
054                    logger.log(level, "SIMON MESSAGE: " + message);
055            }
056    
057            /**
058             * Returns logger used to log messages.
059             *
060             * @return used logger
061             */
062            public Logger getLogger() {
063                    return logger;
064            }
065    
066            /**
067             * Sets the logger that will be used to log messages.
068             *
069             * @param logger new specified logger
070             */
071            public void setLogger(Logger logger) {
072                    this.logger = logger;
073            }
074    
075            /**
076             * Sets the logger by the name of the logger - used by the configure facility to configure the callback.
077             *
078             * @param logger name of the logger
079             */
080            public void setLogger(String logger) {
081                    this.logger = Logger.getLogger(logger);
082            }
083    
084            /**
085             * Returns the specified log level for messages.
086             *
087             * @return log level for messages
088             */
089            public Level getLevel() {
090                    return level;
091            }
092    
093            /**
094             * Sets the level that will be used to log messages.
095             *
096             * @param level log level used to log messages
097             */
098            public void setLevel(Level level) {
099                    this.level = level;
100            }
101    
102            /**
103             * Sets the level via level name - used by the configure facility to configure the callback.
104             *
105             * @param level name of the level
106             */
107            public void setLevel(String level) {
108                    this.level = Level.parse(level);
109            }
110    }
