001    package org.javasimon.console.action;
002    
003    import java.io.IOException;
004    import java.util.Iterator;
005    import javax.servlet.ServletException;
006    import org.javasimon.Sample;
007    import org.javasimon.Simon;
008    import org.javasimon.SimonState;
009    import org.javasimon.console.*;
010    import org.javasimon.console.json.JsonStringifierFactory;
011    import org.javasimon.console.json.ObjectJS;
012    import org.javasimon.console.json.SimpleJS;
013    import org.javasimon.console.text.Stringifier;
014    
015    /**
016     * Base for actions returing data in JSON format
017     *
018     * @author gquintana
019     */
020    public abstract class AbstractJsonAction extends Action {
021    
022            protected AbstractJsonAction(ActionContext context) {
023                    super(context);
024            }
025            /**
026             * Type converter Object &rarr; JSON String
027             */
028            protected final JsonStringifierFactory jsonStringifierFactory=new JsonStringifierFactory();
029            /**
030             * Flag indicating  if simons should be reset during sampling
031             */
032            protected boolean reset;
033            
034            @Override
035            public void readParameters() {
036                    TimeFormatType timeFormat=getContext().getParameterAsEnum("timeFormat", TimeFormatType.class, TimeFormatType.MILLISECOND);
037                    jsonStringifierFactory.init(timeFormat, 
038                            JsonStringifierFactory.READABLE_DATE_PATTERN,// Should be ISO
039                            JsonStringifierFactory.INTEGER_NUMBER_PATTERN
040                            );
041                    reset=getContext().getParameterAsBoolean("reset", Boolean.FALSE);
042            }
043            private <T> void addAttribute(ObjectJS objectJS, String name, Class<T> type, T value) {
044                    objectJS.setAttribute(name, new SimpleJS<T>(value,jsonStringifierFactory.getStringifier(type)));
045            }
046            /**
047             * Transforms a Simon into a JSON object
048             *
049             * @param simon Simon
050             * @return JSON object
051             */
052            protected ObjectJS createObjectJS(Simon simon) {
053                    Sample sample=reset?simon.sampleAndReset():simon.sample();
054                    SimonType lType = SimonTypeFactory.getValueFromInstance(sample);
055                    ObjectJS objectJS = ObjectJS.create(sample, jsonStringifierFactory);
056                    addAttribute(objectJS, "type",          SimonType.class,        lType);
057                    addAttribute(objectJS, "enabled",       Boolean.class,          simon.isEnabled());
058                    addAttribute(objectJS, "state",         SimonState.class,       simon.getState());
059                    return objectJS;
060    
061            }
062            @SuppressWarnings("unchecked")
063            protected final ObjectJS createAttributesObjectJS(Simon simon) {
064                    ObjectJS objectJS = new ObjectJS();
065                    Iterator<String> attributeNameIter=simon.getAttributeNames();
066                    boolean foundAttribute=false;
067                    while(attributeNameIter.hasNext()) {
068                            String attributeName=attributeNameIter.next();
069                            Object attributeValue=simon.getAttribute(attributeName);
070                            if (attributeValue!=null) {
071                                    Stringifier attributeStringifier=jsonStringifierFactory
072                                            .getStringifier(attributeValue.getClass());
073                                    if (attributeStringifier!=null) {
074                                            objectJS.setSimpleAttribute(attributeName, attributeValue, attributeStringifier);
075                                            foundAttribute=true;
076                                    }
077                            }
078    
079                    }
080                    return foundAttribute?objectJS:null;
081            }
082    
083            @Override
084            public void execute() throws ServletException, IOException, ActionException {
085                    dontCache();
086                    getContext().setContentType("application/json");
087            }
088    }
