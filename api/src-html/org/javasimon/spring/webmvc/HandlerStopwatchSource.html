001    package org.javasimon.spring.webmvc;
002    
003    import org.javasimon.Manager;
004    import org.javasimon.source.AbstractStopwatchSource;
005    import org.springframework.web.method.HandlerMethod;
006    
007    /**
008     * Stopwatch source for Spring MVC handlers.
009     *
010     * @author gquintana
011     */
012    public class HandlerStopwatchSource extends AbstractStopwatchSource<HandlerLocation> {
013            /**
014             * Prefix used for Simon names.
015             */
016            private static final String PREFIX = "org.javasimon.mvc";
017    
018            /**
019             * Suffix used for Simon names of Controllers.
020             */
021            private static final String CONTROLLER_SUFFIX = "ctrl";
022    
023            /**
024             * Suffix used for Simon names of Views.
025             */
026            private static final String VIEW_SUFFIX = "view";
027    
028            public HandlerStopwatchSource(Manager manager) {
029                    super(manager);
030            }
031    
032            @Override
033            protected String getMonitorName(HandlerLocation t) {
034                    StringBuilder stringBuilder = new StringBuilder(PREFIX).append(".");
035                    // Append controller type
036                    if (t.getHandler() instanceof HandlerMethod) {
037                            HandlerMethod handlerMethod = (HandlerMethod) t.getHandler();
038                            stringBuilder.append(handlerMethod.getBeanType().getSimpleName())
039                                    .append(".").append(handlerMethod.getMethod().getName());
040                    } else {
041                            stringBuilder.append(t.getHandler().getClass().getSimpleName());
042                    }
043                    // Append step
044                    stringBuilder.append(".");
045                    switch (t.getStep()) {
046                            case CONTROLLER:
047                                    stringBuilder.append(CONTROLLER_SUFFIX);
048                                    break;
049                            case VIEW:
050                                    stringBuilder.append(VIEW_SUFFIX);
051                                    break;
052                    }
053                    return stringBuilder.toString();
054            }
055    }
