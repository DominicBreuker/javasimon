001    package org.javasimon.callback.async;
002    
003    import org.javasimon.proxy.DelegatingMethodInvocation;
004    import org.javasimon.proxy.DelegatingProxyFactory;
005    /**
006     * Proxy factory which can be used make any class asynchronous.
007     * @author gerald
008     * @param <T> 
009     */
010    public class ExecutorProxyFactory<T> extends DelegatingProxyFactory<T> {
011            /**
012             * Executor used for invoking methods on delegate object
013             */
014            private Executor executor;
015            /**
016             * Constructor
017             * @param delegate Delegate object
018             */
019            public ExecutorProxyFactory(T delegate) {
020                    this(delegate, Executors.async());
021            }
022    
023            /**
024             * Constructor
025             * @param delegate Delegate object
026             * @param executor Executor used, see {@link Executors}
027             */
028            public ExecutorProxyFactory(T delegate, Executor executor) {
029                    super(delegate);
030                    this.executor = executor;
031            }
032            /**
033             * Get used executor
034             * @return Executor
035             */
036            public Executor getExecutor() {
037                    return executor;
038            }
039            /**
040             * Set used executor
041             * @param executor Executor
042             */
043            public void setExecutor(Executor executor) {
044                    this.executor = executor;
045            }
046    
047            @Override
048            protected Object invoke(DelegatingMethodInvocation<T> delegatingMethodInvocation) throws Throwable {
049                    return executor.execute(delegatingMethodInvocation);
050            }
051    }
