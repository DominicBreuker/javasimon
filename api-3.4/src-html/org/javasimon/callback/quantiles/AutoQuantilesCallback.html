001    package org.javasimon.callback.quantiles;
002    
003    import java.util.ArrayList;
004    import java.util.List;
005    
006    import org.javasimon.Simon;
007    import org.javasimon.Split;
008    import org.javasimon.Stopwatch;
009    
010    import org.javasimon.utils.SimonUtils;
011    
012    /**
013     * Callback which stores data in buckets to compute quantiles.
014     * Quantiles can only be obtained after warmup period, after which buckets are
015     * initialized. 
016     * For each Simon the following lifecycle occurs:<ol>
017     * <li><em>Warm up</em>:<ul>
018     * <li>Buckets do not exist</li>
019     * <li>Quantiles can not be computed</li>
020     * <li>Splits are kept</li>
021     * </ul>
022     * <li><em>Trigger</em>: splits count >= warmup count threshold<ul>
023     * <li>Buckets are created, configuration (min, max) is determined from kept splits, bucket number is constant</li>
024     * <li>Buckets are filled with previously stored splits</li>
025     * <li>Retained splits are removed</li>
026     * <li>From now on, quantiles can not be computed and splits are not kept anymore</li>
027     * </ul>
028     * <li><em>Normal</em>: <ul>
029     * <li>Buckets are filled/updated with new splits as they come</li>
030     * <li>Quantiles can be computed (provided there is enough splits and buckets are properly configured)</li>
031     * </ul>
032     * </li></ol>
033     *
034     * @author gquintana
035     * @since 3.2
036     * @see Buckets
037     */
038    public class AutoQuantilesCallback extends QuantilesCallback {
039    
040            /**
041             * Simon attribute name of the list of split values stored in Simons before warmup time.
042             */
043            public static final String ATTR_NAME_BUCKETS_VALUES = "bucketsValues";
044    
045            /**
046             * Number of splits before buckets are initialized.
047             * Default 10
048             */
049            private final long warmupCounter;
050    
051            /**
052             * Number of buckets of data for each Simon.
053             */
054            private final int bucketNb;
055    
056            /**
057             * Default constructor.
058             */
059            public AutoQuantilesCallback() {
060                    this.warmupCounter = 10;
061                    this.bucketNb = 8;
062            }
063    
064            /**
065             * Constructor with warmup counter and number of linear buckets for each Simon.
066             */
067            public AutoQuantilesCallback(long warmupCounter, int bucketNb) {
068                    this.warmupCounter = warmupCounter;
069                    this.bucketNb = bucketNb;
070            }
071    
072        /**
073         * Constructor with all configuration
074         * @param bucketsType Linear or exponential
075         * @param warmupCounter Number of splits before init
076         * @param bucketNb Bucket number
077         */
078        public AutoQuantilesCallback(BucketsType bucketsType, long warmupCounter, int bucketNb) {
079            super(bucketsType);
080            this.warmupCounter = warmupCounter;
081            this.bucketNb = bucketNb;
082        }
083    
084        /**
085             * Get the bucket values attribute or create it if it does not exist.
086             */
087            @SuppressWarnings("SynchronizationOnLocalVariableOrMethodParameter")
088            private List<Long> getOrCreateBucketsValues(final Stopwatch stopwatch) {
089                    synchronized (stopwatch) {
090                            List<Long> values = getBucketsValues(stopwatch);
091                            if (values == null) {
092                                    values = new ArrayList<Long>((int) warmupCounter);
093                                    stopwatch.setAttribute(ATTR_NAME_BUCKETS_VALUES, values);
094                            }
095                            return values;
096                    }
097            }
098    
099            /**
100             * Get the bucket values attribute.
101             */
102            @SuppressWarnings("unchecked")
103            private List<Long> getBucketsValues(final Stopwatch stopwatch) {
104                    return (List<Long>) stopwatch.getAttribute(ATTR_NAME_BUCKETS_VALUES);
105            }
106    
107    
108            /**
109             * Remove the bucket values attribute (after warmup).
110             */
111            private void removeBucketsValues(final Stopwatch stopwatch) {
112                    stopwatch.removeAttribute(ATTR_NAME_BUCKETS_VALUES);
113            }
114    
115            /**
116             * Create the buckets after warmup time.
117             * Can be overridden to customize buckets configuration.
118             * By default buckets are create with:<ul>
119             * <li>Min: stopwatch min-10% rounded to inferior millisecond</li>
120             * <li>Max: stopwatch max+10 rounded to superior millisecond</li>
121             * <li>Nb buckets: {@link #bucketNb}
122             *
123             * @param stopwatch Stopwatch (containing configuration)
124             * @return new Buckets objects
125             */
126            protected Buckets createBucketsAfterWarmup(Stopwatch stopwatch) {
127                    // Compute min
128                    long min = stopwatch.getMin() * 90L / 100L; // min -10%
129                    min = Math.max(0, min); // no negative mins
130                    min = (min / SimonUtils.NANOS_IN_MILLIS) * SimonUtils.NANOS_IN_MILLIS; // round to lower millisecond
131                    // Compute max
132                    long max = (stopwatch.getMax() * 110L) / 100L; // max +10%
133                    max = (max / SimonUtils.NANOS_IN_MILLIS + 1) * SimonUtils.NANOS_IN_MILLIS; // round to upper millisecond
134                    return createBuckets(stopwatch, min, max, bucketNb);
135            }
136            /**
137             * When warmup ends, buckets are create and retained splits are
138             * sorted in the buckets.
139             */
140            protected final Buckets createBuckets(Stopwatch stopwatch) {
141                    if (stopwatch.getCounter() > warmupCounter) {
142                            Buckets buckets = createBucketsAfterWarmup(stopwatch);
143                            // Add retained splits to buckets
144                            buckets.addValues(getBucketsValues(stopwatch));
145                            removeBucketsValues(stopwatch);
146                            return buckets;
147                    } else {
148                            return null;
149                    }
150            }
151    
152            /**
153             * When simon is created, the list containing Split values is added to stopwatch attributes.
154             */
155            @Override
156            public void onSimonCreated(Simon simon) {
157                    if (simon instanceof Stopwatch) {
158                            Stopwatch stopwatch = (Stopwatch) simon;
159                            getOrCreateBucketsValues(stopwatch);
160                    }
161            }
162    
163            /**
164             * Called when there is a new split on a Stopwatch, either
165             * {@link #onStopwatchStop} or {@link #onStopwatchAdd}.
166             * If buckets have been initialized, the value is added to appropriate bucket.
167             * Else if stopwatch is warming up value is added to value list.
168             */
169            @Override
170            protected void onStopwatchSplit(Stopwatch stopwatch, Split split) {
171                    Buckets buckets = getOrCreateBuckets(stopwatch);
172                    long value = split.runningFor();
173                    if (buckets == null) {
174                            // Warming up
175                            getOrCreateBucketsValues(stopwatch).add(value);
176                    } else {
177                            // Warm
178                            buckets.addValue(value);
179                            buckets.log(split);
180                    }
181            }
182    
183            /**
184             * Called when there is a new split on a Stopwatch, either
185             * {@link #onStopwatchStop} or {@link #onStopwatchAdd}.
186             * If buckets have been initialized, the value is added to appropriate bucket.
187             * Else if stopwatch is warming up value is added to value list.
188             */
189            @Override
190            protected void onStopwatchSplit(Stopwatch stopwatch, long value) {
191                    Buckets buckets = getOrCreateBuckets(stopwatch);
192                    if (buckets == null) {
193                            // Warming up
194                            getOrCreateBucketsValues(stopwatch).add(value);
195                    } else {
196                            // Warm
197                            buckets.addValue(value);
198                    }
199            }
200            
201            /**
202             * When the Stopwatch is reset, so are the buckets.
203             */
204            @Override
205            public void onSimonReset(Simon simon) {
206                    if (simon instanceof Stopwatch) {
207                            Stopwatch stopwatch = (Stopwatch) simon;
208                            Buckets buckets = getBuckets(stopwatch);
209                            if (buckets == null) {
210                                    // Cold
211                                    List<Long> values = getBucketsValues(stopwatch);
212                                    if (values != null) {
213                                            // Warming up
214                                            values.clear();
215                                    }
216                            } else {
217                                    // Warm
218                                    buckets.clear();
219                            }
220                    }
221            }
222    }
