001    package org.javasimon.console.action;
002    
003    import java.io.IOException;
004    import javax.servlet.ServletException;
005    import org.javasimon.Simon;
006    import org.javasimon.console.ActionContext;
007    import org.javasimon.console.ActionException;
008    import org.javasimon.console.json.ArrayJS;
009    import org.javasimon.console.json.ObjectJS;
010    
011    /**
012     * Export Simons as a hierarchical JSON object for display in a treetable.
013     * All attributes of simons are exported
014     *
015     * @author gquintana
016     */
017    public class TreeJsonAction extends AbstractJsonAction {
018    
019            public static final String PATH = "/data/tree.json";
020            /**
021             * Name of the simon from where to start.
022             * {@code null} means root.
023             */
024            private String name;
025            public TreeJsonAction(ActionContext context) {
026                    super(context);
027            }
028    
029            @Override
030            public void readParameters() {
031                    super.readParameters();
032                    name = getContext().getParameterAsString("name", null);
033            }
034    
035            @Override
036            protected ObjectJS createObjectJS(Simon simon) {
037                    ObjectJS simonJS = super.createObjectJS(simon);
038                    ArrayJS childrenJS = new ArrayJS();
039                    for (Simon child : simon.getChildren()) {
040                            ObjectJS childJS = createObjectJS(child);
041                            childrenJS.addElement(childJS);
042                    }
043                    simonJS.setAttribute("children", childrenJS);
044                    return simonJS;
045            }
046    
047            @Override
048            public void execute() throws ServletException, IOException, ActionException {
049                    getContext().setContentType("application/json");
050                    Simon simon = name == null ? getContext().getManager().getRootSimon() : getContext().getManager().getSimon(name);
051                    ObjectJS simonRootJS = createObjectJS(simon);
052                    simonRootJS.write(getContext().getWriter());
053            }
054    }
