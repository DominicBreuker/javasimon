001    package org.javasimon.javaee;
002    
003    import javax.servlet.http.HttpServletRequest;
004    
005    import org.javasimon.Manager;
006    import org.javasimon.Stopwatch;
007    import org.javasimon.source.AbstractStopwatchSource;
008    import org.javasimon.source.CachedStopwatchSource;
009    import org.javasimon.source.StopwatchSource;
010    import org.javasimon.utils.Replacer;
011    
012    /**
013     * Provide stopwatch source for HTTP Servlet request.
014     * Used by {@link SimonServletFilter} as default stopwatch source.
015     * Can be overridden to customize monitored HTTP Requests and their
016     * related Simon name.
017     * <p/>
018     * To select which HTTP Request should be monitored method {@link #isMonitored} can be overridden. Default implementation monitors everything except for
019     * typical resource-like requests (images, JS/CSS, ...).
020     *
021     * @author gquintana
022     * @author <a href="mailto:virgo47@gmail.com">Richard "Virgo" Richter</a>
023     */
024    public class HttpStopwatchSource extends AbstractStopwatchSource<HttpServletRequest> {
025            /**
026             * Default prefix for web filter Simons if no "prefix" init parameter is used.
027             */
028            public static final String DEFAULT_SIMON_PREFIX = "org.javasimon.web";
029    
030            /**
031             * Simon prefix, can be set to {@code null}.
032             */
033            private String prefix = DEFAULT_SIMON_PREFIX;
034    
035            private Replacer unallowedCharacterReplacer = SimonServletFilterUtils.createUnallowedCharsReplacer("_");
036            private Replacer jsessionParameterReplacer = new Replacer("[;&]?JSESSIONID=[^;?/&]*", "", Replacer.Modificator.IGNORE_CASE);
037            private Replacer trailingStuffReplacer = new Replacer("/[^a-zA-Z]*$", "");
038    
039            public HttpStopwatchSource(Manager manager) {
040                    super(manager);
041            }
042    
043            public String getPrefix() {
044                    return prefix;
045            }
046    
047            public void setPrefix(String prefix) {
048                    this.prefix = prefix;
049            }
050    
051            public String getReplaceUnallowed() {
052                    return unallowedCharacterReplacer.getTo();
053            }
054    
055            public void setReplaceUnallowed(String replaceUnallowed) {
056                    unallowedCharacterReplacer.setTo(replaceUnallowed);
057            }
058    
059            /**
060             * Returns Simon name for the specified HTTP request with the specified prefix. By default it contains URI without parameters with
061             * all slashes replaced for dots (slashes then determines position in Simon hierarchy). Method can NOT be overridden, but some of the
062             * following steps can:
063             * <ol>
064             * <li>the request is transformed to the string ({@link #requestToStringForMonitorName(javax.servlet.http.HttpServletRequest)}, can be overridden),</li>
065             * <li>the characters that are not allowed as part of the Simon name are replaced with underscore (_) - replacement regex can be changed with {@link #setReplaceUnallowed(String)},</li>
066             * <li>any subsequent slashes and dots are replaced with a single dot ({@link org.javasimon.Manager#HIERARCHY_DELIMITER})</li>
067             * </ol>
068             *
069             * @param request HTTP request
070             * @return fully qualified name of the Simon
071             * @see #requestToStringForMonitorName(javax.servlet.http.HttpServletRequest)
072             */
073            protected final String getMonitorName(HttpServletRequest request) {
074                    String uri = requestToStringForMonitorName(request);
075                    String localName = SimonServletFilterUtils.getSimonName(uri, unallowedCharacterReplacer);
076                    if (prefix == null || prefix.isEmpty()) {
077                            return localName;
078                    }
079                    return prefix + Manager.HIERARCHY_DELIMITER + localName;
080            }
081    
082            /**
083             * Performs the first step in getting the monitor name from the specified HTTP request - here any custom ignore logic should happen.
084             * By default the name is URI (without parameters - see {@link javax.servlet.http.HttpServletRequest#getRequestURI()}) with JSessionID
085             * removed (see {@link #removeJSessionIdFromUri(String)}) and any trailing stuff removed (see {@link #removeTrailingStuff(String)}).
086             * This method can be overridden for two typical reasons:
087             * <ul>
088             * <li>Name of the monitor (Stopwatch) should be based on something else then URI,</li>
089             * <li>there are other parts of the name that should be modified or ignored (e.g., REST parameters that are part of the URI).</li>
090             * </ul>
091             *
092             * @param request HTTP request
093             * @return preprocessed URI that will be converted to the Simon name
094             * @see #getMonitorName(javax.servlet.http.HttpServletRequest)
095             * @see #removeJSessionIdFromUri(String)
096             */
097            protected String requestToStringForMonitorName(HttpServletRequest request) {
098                    String uri = request.getRequestURI();
099                    uri = removeJSessionIdFromUri(uri);
100                    uri = removeTrailingStuff(uri);
101                    return uri;
102            }
103    
104            /**
105             * Removes JSESSIONID parameter from URI. By default it is not necessary to handle parameters, as incoming URI already is without
106             * parameters, but JSESSIONID sometimes come before parameters in other forms and this method tries to remove such forms.
107             * <p/>
108             * Called by default implementation of {@link #requestToStringForMonitorName(javax.servlet.http.HttpServletRequest)} and extracted
109             * so it can be used by any overriding implementation of the same method. Method can be overridden if the default behavior is not
110             * sufficient.
111             *
112             * @param uri preprocessed URI that may contain JSessionID
113             * @return preprocessed URI without JSessionID
114             * @see #requestToStringForMonitorName(javax.servlet.http.HttpServletRequest)
115             */
116            protected String removeJSessionIdFromUri(String uri) {
117                    return jsessionParameterReplacer.process(uri);
118            }
119    
120            /**
121             * Removes any trailing slashes followed by other characters if none of them is alphabetic. This should take care of some REST
122             * parameters (numeric id-s) and it also removes trailing slashes to avoid empty local Simon names which is forbidden.
123             * <p/>
124             * Called by default implementation of {@link #requestToStringForMonitorName(javax.servlet.http.HttpServletRequest)} and extracted
125             * so it can be used by any overriding implementation of the same method. Method can be overridden if the default behavior is not
126             * sufficient.
127             *
128             * @param uri preprocessed URI that may contain JSessionID
129             * @return preprocessed URI without JSessionID
130             * @see #requestToStringForMonitorName(javax.servlet.http.HttpServletRequest)
131             */
132            protected String removeTrailingStuff(String uri) {
133                    return trailingStuffReplacer.process(uri);
134            }
135    
136            /**
137             * Indicates whether the HTTP Request should be monitored - method is intended for override.
138             * Default behavior ignores URIs ending with .css, .png, .gif, .jpg and .js (ignores casing).
139             *
140             * @param httpServletRequest HTTP Request
141             * @return true to enable request monitoring, false either
142             */
143            @Override
144            public boolean isMonitored(HttpServletRequest httpServletRequest) {
145                    String uri = httpServletRequest.getRequestURI().toLowerCase();
146                    return !(uri.endsWith(".css") || uri.endsWith(".png") || uri.endsWith(".gif") || uri.endsWith(".jpg") || uri.endsWith(".js"));
147            }
148    
149            /**
150             * Get a stopwatch for given HTTP request.
151             *
152             * @param request Method HTTP request
153             * @return Stopwatch for the HTTP request
154             */
155            @Override
156            public Stopwatch getMonitor(HttpServletRequest request) {
157                    final Stopwatch stopwatch = super.getMonitor(request);
158                    if (stopwatch.getNote() == null) {
159                            stopwatch.setNote(request.getRequestURI());
160                    }
161                    return stopwatch;
162            }
163    
164            /**
165             * Wraps given stop watch source in a cache.
166             *
167             * @param stopwatchSource Stopwatch source
168             * @return Cached stopwatch source
169             */
170            public static StopwatchSource<HttpServletRequest> newCacheStopwatchSource(StopwatchSource<HttpServletRequest> stopwatchSource) {
171                    return new CachedStopwatchSource<HttpServletRequest, String>(stopwatchSource) {
172                            @Override
173                            protected String getLocationKey(HttpServletRequest location) {
174                                    return location.getRequestURI();
175                            }
176                    };
177            }
178    }
