001    package org.javasimon.source;
002    
003    import org.javasimon.Manager;
004    import org.javasimon.Simon;
005    
006    import java.util.HashMap;
007    import java.util.Map;
008    
009    /**
010     * Monitor source playing the role of cache for delegate monitor source.
011     *
012     * @param <L> Location/invocation context
013     * @param <M> Simon type
014     * @param <K> Location key
015     * @author gquintana
016     */
017    public abstract class CachedMonitorSource<L, M extends Simon, K> implements MonitorSource<L, M> {
018            /**
019             * Real monitor source.
020             */
021            private final MonitorSource<L, M> delegate;
022    
023            /**
024             * Monitor/location information.
025             */
026            private static class MonitorInformation {
027                    private final boolean monitored;
028                    private final String name;
029    
030                    public MonitorInformation(boolean monitored, Simon simon) {
031                            this.monitored = monitored;
032                            if (simon == null) {
033                                    name = null;
034                            } else {
035                                    this.name = simon.getName();
036                            }
037                    }
038    
039                    public boolean isMonitored() {
040                            return monitored;
041                    }
042    
043                    public String getName() {
044                            return name;
045                    }
046    
047                    public Simon getMonitor(Manager manager) {
048                            if (name == null) {
049                                    return null;
050                            } else {
051                                    return manager.getSimon(name);
052                            }
053                    }
054            }
055    
056            /**
057             * Not monitored monitor information.
058             */
059            private static final MonitorInformation NULL_MONITOR_INFORMATION = new MonitorInformation(false, null);
060    
061            /**
062             * Map location key &rarr; monitor information.
063             */
064            private final Map<K, MonitorInformation> monitorInformations = new HashMap<K, MonitorInformation>();
065    
066            /**
067             * Constructor with real {@link MonitorSource}.
068             *
069             * @param delegate Delegate provider monitors for real
070             */
071            public CachedMonitorSource(MonitorSource<L, M> delegate) {
072                    this.delegate = delegate;
073            }
074    
075            /**
076             * Get location for given location.
077             */
078            protected abstract K getLocationKey(L location);
079    
080            /**
081             * Get monitor information for given location.
082             * First monitor information is looked up in cache.
083             * Then, when not found, delegate is called.
084             *
085             * @param location Location
086             * @return Monitor information
087             */
088            private MonitorInformation getMonitorInformation(L location) {
089                    final K monitorKey = getLocationKey(location);
090                    MonitorInformation monitorInformation = monitorInformations.get(monitorKey);
091                    if (monitorInformation == null) {
092                            // Not found, let's call delegate
093                            if (delegate.isMonitored(location)) {
094                                    monitorInformation = new MonitorInformation(true, delegate.getMonitor(location));
095                            } else {
096                                    monitorInformation = NULL_MONITOR_INFORMATION;
097                            }
098                            monitorInformations.put(monitorKey, monitorInformation);
099                    }
100                    return monitorInformation;
101            }
102    
103            /**
104             * Remove monitor information for given location.
105             */
106            private void removeMonitorInformation(L location) {
107                    monitorInformations.remove(getLocationKey(location));
108            }
109    
110            /**
111             * Check whether location should be monitored.
112             * Response is entirely based on cache
113             */
114            @Override
115            public boolean isMonitored(L location) {
116                    return getMonitorInformation(location).isMonitored();
117            }
118    
119            @SuppressWarnings("unchecked")
120            private M getMonitorOnce(L location) {
121                    return (M) getMonitorInformation(location).getMonitor(getManager());
122            }
123    
124            /**
125             * Get Simon for the specified location.
126             * Simon is retrieved from name in cache.
127             *
128             * @param location Location
129             * @return Simon for the specified location
130             */
131            @Override
132            public M getMonitor(L location) {
133                    M monitor = getMonitorOnce(location);
134                    // In case monitor was removed from manager, we retry
135                    if (monitor == null) {
136                            removeMonitorInformation(location);
137                            monitor = getMonitorOnce(location);
138                    }
139                    return monitor;
140            }
141    
142            @Override
143            public Manager getManager() {
144                    return delegate.getManager();
145            }
146    }
