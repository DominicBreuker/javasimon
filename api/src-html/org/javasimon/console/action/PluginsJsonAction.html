001    package org.javasimon.console.action;
002    
003    import java.io.IOException;
004    import java.util.Collections;
005    import java.util.List;
006    import javax.servlet.ServletException;
007    import org.javasimon.console.ActionContext;
008    import org.javasimon.console.ActionException;
009    import org.javasimon.console.SimonConsolePlugin;
010    import org.javasimon.console.json.ArrayJS;
011    
012    /**
013     * Action to get plugin list in JSON format.
014     * @author gquintana
015     */
016    public class PluginsJsonAction extends AbstractJsonAction {
017            public static final String PATH="/data/plugins.json";
018            /**
019             * Expected plugin type
020             */
021            private Class<? extends SimonConsolePlugin> type;
022            private boolean invalidType=false;
023            
024            public PluginsJsonAction(ActionContext context) {
025                    super(context);
026            }
027            @Override
028            public void readParameters() {
029                    super.readParameters();
030                    String sType=getContext().getParameterAsString("type", null);
031                    if (sType!=null) {
032                            try {
033                                    Class<?> oType = Class.forName(sType);
034                                    if (SimonConsolePlugin.class.isAssignableFrom(oType)) {
035                                            type = (Class<? extends SimonConsolePlugin>) oType;
036                                    } else {
037                                            invalidType = true;
038                                    }
039                            } catch (Exception e) {
040                                    invalidType = true;
041                            }
042                    }
043                    
044            }
045    
046            @Override
047            public void execute() throws ServletException, IOException, ActionException {
048                    super.execute();
049                    // Get plugin list
050                    List<? extends SimonConsolePlugin> plugins;
051                    if (invalidType) {
052                            plugins=Collections.emptyList();
053                    } else if (type==null) {
054                            plugins=getContext().getPluginManager().getPlugins();
055                    } else {
056                            plugins=getContext().getPluginManager().getPluginsByType(type);
057                    }
058                    // Convert list to JSON
059                    ArrayJS pluginsJS=new ArrayJS();
060                    for(SimonConsolePlugin plugin:plugins) {
061                            pluginsJS.addElement(plugin.toJson(jsonStringifierFactory));
062                    }
063                    pluginsJS.write(getContext().getWriter());
064            }
065    
066            
067    }
