013    /**
014     * SimonUtils provides static utility methods.
015     * <h3>Human readable outputs</h3>
016     * Both {@link org.javasimon.Stopwatch} and {@link org.javasimon.Counter} provide human readable
017     * {@code toString} outputs. All nanosecond values are converted into few valid digits with
018     * proper unit (ns, us, ms, s) - this is done via method {@link #presentNanoTime(long)}.
019     * Max/min counter values are checked for undefined state (max/min long value is converted
020     * to string "undef") - via method {@link #presentMinMaxCount(long)}.
021     * <h3>Simon tree operations</h3>
022     * Method for recursive reset of the {@link org.javasimon.Simon} and all its children is provided -
023     * {@link #recursiveReset(org.javasimon.Simon)}. For various debug purposes there is a method
024     * that creates string displaying the whole Simon sub-tree. Here is example code that initializes
025     * two random Simons and prints the whole Simon hierarchy (note that the method can be used to
026     * obtain any sub-tree of the hierarchy):
027     * <pre>
028     * Split split = SimonManager.getStopwatch("com.my.other.stopwatch").start();
029     * SimonManager.getCounter("com.my.counter").setState(SimonState.DISABLED, false);
030     * split.stop();
031     * System.out.println(SimonUtils.simonTreeString(SimonManager.getRootSimon()));</pre>
032     * And the output is:
033     * <pre>
034     * (+): Unknown Simon:  [ ENABLED]
035     *   com(+): Unknown Simon:  [com INHERIT]
036     *     my(+): Unknown Simon:  [com.my INHERIT]
037     *       other(+): Unknown Simon:  [com.my.other INHERIT]
038     *         stopwatch(+): Simon Stopwatch: total 24.2 ms, counter 1, max 24.2 ms, min 24.2 ms, mean 24.2 ms [com.my.other.stopwatch INHERIT]
039     *       counter(-): Simon Counter: counter=0, max=undef, min=undef [com.my.counter DISABLED]</pre>
040     * Notice +/- signs in parenthesis that displays effective Simon state (enabled/disabled), further
041     * details are printed via each Simon's {@code toString} method.
042     * <h3>Other utilities</h3>
043     * It is possible to obtain "local name" of the Simon (behind the last dot) via {@link #localName(String)}
044     * or check if the name is valid Simon name via {@link #checkName(String)}.
045     *
046     * @author <a href="mailto:virgo47@gmail.com">Richard "Virgo" Richter</a>
047     * @author Radovan Sninsky
048     * @since 1.0
049     */

109            private static final DecimalFormatSymbols DECIMAL_FORMAT_SYMBOLS = new DecimalFormatSymbols(Locale.US);

119            static {
120                    // Finds out the index of "this code" in the returned stack trace - funny but it differs in JDK 1.5 and 1.6
121                    int i = 1;
122                    for (StackTraceElement ste : Thread.currentThread().getStackTrace()) {
123                            i++;
124                            if (ste.getClassName().equals(SimonUtils.class.getName())) {
125                                    break;
126                            }
127                    }
128                    CLIENT_CODE_STACK_INDEX = i;
129    
130                    // for conversion between nano and millis - see method millisForNano(long)
131                    INIT_NANOS = System.nanoTime();
132                    INIT_MILLIS = System.currentTimeMillis();
133            }

139            /**
140             * Returns nano-time in human readable form with unit. Number is always from 10 to 9999
141             * except for seconds that are the biggest unit used.
142             *
143             * @param nanos time in nanoseconds
144             * @return human readable time string
145             */
146            public static String presentNanoTime(long nanos) {
147                    if (nanos == Long.MAX_VALUE) {
148                            return UNDEF_STRING;
149                    }
150                    return presentOverNanoTime((double) nanos);
151            }
152    
153            /**
154             * Returns nano-time in human readable form with unit. Number is always from 10 to 9999
155             * except for seconds that are the biggest unit used.
156             *
157             * @param nanos time in nanoseconds
158             * @return human readable time string
159             */
160            public static String presentNanoTime(double nanos) {
161                    if (nanos == Double.MAX_VALUE) {
162                            return UNDEF_STRING;
163                    }
164                    return presentOverNanoTime(nanos);
165            }
166    
167            /**
168             * @param time time in nanoseconds
169             * @return human readable time string
170             */
171            private static String presentOverNanoTime(double time) {
172                    if (Math.abs(time) < 1d) {
173                            return "0";
174                    }
175    
176                    if (time < UNIT_PREFIX_FACTOR) {
177                            return ((long) time) + " ns";
178                    }
179    
180                    time /= UNIT_PREFIX_FACTOR;
181                    if (time < UNIT_PREFIX_FACTOR) {
182                            return formatTime(time, " us");
183                    }
184    
185                    time /= UNIT_PREFIX_FACTOR;
186                    if (time < UNIT_PREFIX_FACTOR) {
187                            return formatTime(time, " ms");
188                    }
189    
190                    time /= UNIT_PREFIX_FACTOR;
191                    return formatTime(time, " s");
192            }
193    
194            private static String formatTime(double time, String unit) {
195                    if (time < TEN) {
196                            synchronized (UNDER_TEN_FORMAT) {
197                                    return UNDER_TEN_FORMAT.format(time) + unit;
198                            }
199                    }
200                    if (time < HUNDRED) {
201                            synchronized (UNDER_HUNDRED_FORMAT) {
202                                    return UNDER_HUNDRED_FORMAT.format(time) + unit;
203                            }
204                    }
205                    synchronized (DEFAULT_FORMAT) {
206                            return DEFAULT_FORMAT.format(time) + unit;
207                    }
208            }

210            /**
211             * Returns timestamp in human readable (yet condensed) form "yyMMdd-HHmmss.SSS".
212             *
213             * @param timestamp timestamp in millis
214             * @return timestamp as a human readable string
215             */
216            public static String presentTimestamp(long timestamp) {
217                    if (timestamp == 0) {
218                            return UNDEF_STRING;
219                    }
220                    synchronized (TIMESTAMP_FORMAT) {
221                            return TIMESTAMP_FORMAT.format(new Date(timestamp));
222                    }
223            }

226             * Returns min/max counter values in human readable form - if the value is max or min long value
227             * it is considered unused and string "undef" is returned.
228             *
229             * @param minmax counter extreme value
230             * @return counter value or "undef" if counter contains {@code Long.MIN_VALUE} or {@code Long.MAX_VALUE}

233                    if (minmax == Long.MAX_VALUE || minmax == Long.MIN_VALUE) {

253            /**
254             * Returns multi-line string containing Simon tree starting with the specified Simon.
255             * Root Simon can be used to obtain tree with all Simons. Returns {@code null} for
256             * input value of null or for NullSimon or any Simon with name equal to null (anonymous
257             * Simons) - this is also the case when the Manager is disabled and tree for its root
258             * Simon is requested.
259             *
260             * @param simon root Simon of the output tree
261             * @return string containing the tree or null if the Simon is null Simon
262             */
263            public static String simonTreeString(Simon simon) {
264                    if (simon == null || simon.getName() == null) {
265                            return null;
266                    }
267                    StringBuilder sb = new StringBuilder();
268                    printSimonTree(0, simon, sb);
269                    return sb.toString();
270            }
271    
272            private static void printSimonTree(int level, Simon simon, StringBuilder sb) {
273                    printSimon(level, simon, sb);
274                    for (Simon child : simon.getChildren()) {
275                            printSimonTree(level + 1, child, sb);
276                    }
277            }
278    
279            private static void printSimon(int level, Simon simon, StringBuilder sb) {
280                    for (int i = 0; i < level; i++) {
281                            sb.append("  ");
282                    }
283                    sb.append(localName(simon.getName()))
284                            .append('(')
285                            .append(simon.isEnabled() ? '+' : '-')
286                            .append("): ")
287                            .append(simon.toString())
288                            .append('\n');
289            }

291            /**
292             * Returns last part of Simon name - local name.
293             *
294             * @param name full Simon name
295             * @return string containing local name
296             */
297            public static String localName(String name) {
298                    int ix = name.lastIndexOf(Manager.HIERARCHY_DELIMITER);
299                    if (ix == -1) {
300                            return name;
301                    }
302                    return name.substring(ix + 1);
303            }

305            /**
306             * Resets the whole Simon subtree - calls {@link org.javasimon.Simon#reset()} on the
307             * Simon and recursively on all its children. Operation is not truly atomic as a whole,
308             * consistency on the Simon level depends on the implementation of {@link org.javasimon.Simon#reset()}
309             * (which is thread-safe in all current implementations).
310             *
311             * @param simon subtree root
312             */
313            public static void recursiveReset(Simon simon) {
314                    simon.reset();
315                    for (Simon child : simon.getChildren()) {
316                            recursiveReset(child);
317                    }
318            }

345             * Autogenerates name for the Simon using the fully qualified class name and the method name.
346             * <i>This method has inherent performance penalty (getting the stack-trace) so it is recommended to store
347             * pregenerated name. (Whole get-start-stop cycle is roughly 30-times slower with this method included.)</i>

349             * @param suffix name suffix for eventual Simon discrimination

353            public static String generateNameForClassAndMethod(String suffix) {
354                    return generatePrivate(suffix, true);
355            }

357            // method is exatracted, so the stack trace index is always right
358            private static String generatePrivate(String suffix, boolean includeMethodName) {
359                    StackTraceElement stackElement = Thread.currentThread().getStackTrace()[CLIENT_CODE_STACK_INDEX];
360                    StringBuilder nameBuilder = new StringBuilder(stackElement.getClassName());
361                    if (includeMethodName) {
362                            nameBuilder.append('.').append(stackElement.getMethodName());
363                    }
364                    if (suffix != null) {
365                            nameBuilder.append(suffix);
366                    }
367                    return nameBuilder.toString();
368            }

377                    return generatePrivate(null, true);

380            /**
381             * Calls a block of code with stopwatch around and returns result.
382             *
383             * @param name name of the Stopwatch
384             * @param callable callable block of code
385             * @param <T> return type
386             * @return whatever block of code returns
387             * @throws Exception whatever block of code throws
388             * @since 3.0
389             */
390            public static <T> T doWithStopwatch(String name, Callable<T> callable) throws Exception {
391                    Split split = SimonManager.getStopwatch(name).start();
392                    try {
393                            return callable.call();
394                    } finally {
395                            split.stop();
396                    }
397            }
398    
399            /**
400             * Calls a block of code with stopwatch around, can not return any result or throw an exception
401             * (use {@link #doWithStopwatch(String, java.util.concurrent.Callable)} instead).
402             *
403             * @param name name of the Stopwatch
404             * @param runnable wrapped block of code
405             * @since 3.0
406             */
407            public static void doWithStopwatch(String name, Runnable runnable) {
408                    Split split = SimonManager.getStopwatch(name).start();
409                    try {
410                            runnable.run();
411                    } finally {
412                            split.stop();
413                    }
414            }

416            private static final String SHRINKED_STRING = "...";
417    
418            /**
419             * Shrinks the middle of the input string if it is too long, so it does not exceed limitTo.
420             *
421             * @since 3.2
422             */
423            public static String compact(String input, int limitTo) {
424                    if (input == null || input.length() <= limitTo) {
425                            return input;
426                    }
427    
428                    int headLength = limitTo / 2;
429                    int tailLength = limitTo - SHRINKED_STRING.length() - headLength;
430                    if (tailLength < 0) {
431                            tailLength = 1;
432                    }
433    
434                    return input.substring(0, headLength) + SHRINKED_STRING + input.substring(input.length() - tailLength);
435            }

437            /**
438             * Converts nano timer value into millis timestamp compatible with {@link System#currentTimeMillis()}. Method does not
439             * just divide nanos by one million, but also works with remembered values for milli- and nano-timers at one particular moment.
440             *
441             * @param nanos nano timer value
442             * @return ms timestamp
443             * @since 3.3 (moved from SimonManager where it was since 3.1)
444             */
445            public static long millisForNano(long nanos) {
446                    return INIT_MILLIS + (nanos - INIT_NANOS) / NANOS_IN_MILLIS;
447            }
