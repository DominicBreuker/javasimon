001    package org.javasimon.console;
002    
003    import javax.servlet.*;
004    import javax.servlet.http.HttpServletRequest;
005    import javax.servlet.http.HttpServletResponse;
006    import java.io.IOException;
007    
008    import org.javasimon.Manager;
009    import org.javasimon.utils.SimonUtils;
010    
011    /**
012     * Simon Console filter does the same as {@link SimonConsoleServlet} and provides just an alternative way
013     * to include embeddable console in your applications. Putting this filter before the Simon Servlet filter is also
014     * the easiest way how to avoid monitoring of the console itself.
015     *
016     * @author Richard Richter
017     * @since 2.3
018     */
019    @SuppressWarnings("UnusedParameters")
020    public class SimonConsoleFilter implements Filter {
021            private SimonConsoleRequestProcessor requestProcessor;
022    
023            @Override
024            public final void init(FilterConfig filterConfig) {
025                    requestProcessor = new SimonConsoleRequestProcessor(
026                            filterConfig.getInitParameter(SimonConsoleServlet.URL_PREFIX_INIT_PARAMETER));
027                    requestProcessor.initActionBindings();
028                    pickUpSharedManagerIfExists(filterConfig);
029            }
030    
031            private void pickUpSharedManagerIfExists(FilterConfig filterConfig) {
032                    Object managerObject = filterConfig.getServletContext().getAttribute(SimonUtils.MANAGER_SERVLET_CTX_ATTRIBUTE);
033                    if (managerObject != null && managerObject instanceof Manager) {
034                            requestProcessor.setManager((Manager) managerObject);
035                    }
036            }
037    
038            /**
039             * Wraps the HTTP request with Simon measuring. Separate Simons are created for different URIs (parameters
040             * ignored).
041             *
042             * @param servletRequest HTTP servlet request
043             * @param servletResponse HTTP servlet response
044             * @param filterChain filter chain
045             * @throws java.io.IOException possibly thrown by other filter/serlvet in the chain
046             * @throws javax.servlet.ServletException possibly thrown by other filter/serlvet in the chain
047             */
048            public final void doFilter(ServletRequest servletRequest, ServletResponse servletResponse, FilterChain filterChain) throws IOException, ServletException {
049                    HttpServletRequest request = (HttpServletRequest) servletRequest;
050                    HttpServletResponse response = (HttpServletResponse) servletResponse;
051    
052                    String localPath = request.getRequestURI().substring(request.getContextPath().length());
053                    if (localPath.startsWith(requestProcessor.getUrlPrefix())) {
054                            requestProcessor.processRequest(request, response);
055                            return;
056                    }
057    
058                    filterChain.doFilter(request, response);
059            }
060    
061            @Override
062            public void destroy() {
063            }
064    }
