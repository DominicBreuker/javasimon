001    package org.javasimon;
002    
003    import java.util.Iterator;
004    import java.util.Map;
005    
006    import org.javasimon.utils.SimonUtils;
007    
008    /**
009     * Represents single time split - one Stopwatch measurement. Object is obtained by {@link org.javasimon.Stopwatch#start()}
010     * and the measurement is ended using {@link #stop()} method on this object. Split will return 0 as the result
011     * if the related Stopwatch was disabled when the Split was obtained. The Split can be stopped in any other thread.
012     * Split measures real time (based on {@link System#nanoTime()}), it does not measure CPU time. Split can be garbage collected
013     * and no resource leak occurs if it is not stopped, however Stopwatch's active counter ({@link org.javasimon.Stopwatch#getActive()})
014     * will be stay incremented.
015     * <p/>
016     * Split can never be running ({@link #isRunning()}) if it is disabled. Enabled split is running until it is stopped.
017     * Stopped split (not running) will never again be running. Split never changes enabled flag after creation.
018     *
019     * @author <a href="mailto:virgo47@gmail.com">Richard "Virgo" Richter</a>
020     * @see Stopwatch
021     */
022    public final class Split implements HasAttributes {
023            /**
024             * Disabled split (implies not running) for cases where monitoring is disabled and {@code null} value is not an option.
025             */
026            public static final Split DISABLED = new Split(false);
027    
028            /**
029             * Attribute name under which effectively used stopwatch is stored if the split was stopped with {@link #stop(String)}.
030             */
031            public static final String ATTR_EFFECTIVE_STOPWATCH = "effective-stopwatch";
032    
033            private volatile Stopwatch stopwatch;
034            private final boolean enabled;
035            private volatile boolean running;
036    
037            private volatile long start;
038            private volatile long total;
039    
040            private AttributesSupport attributesSupport = new AttributesSupport();
041    
042            private Split(boolean enabled) {
043                    this.enabled = enabled;
044            }
045    
046            /**
047             * Creates a new Split for an enabled Stopwatch with a specific timestamp in nanoseconds - <b>called internally only</b>.
048             *
049             * @param stopwatch owning Stopwatch (enabled)
050             * @param start start timestamp in nanoseconds
051             */
052            Split(Stopwatch stopwatch, long start) {
053                    assert start > 0 : "start ns value should not be 0 in this constructor!";
054    
055                    this.stopwatch = stopwatch;
056                    this.start = start;
057                    enabled = true;
058                    running = true;
059            }
060    
061            /**
062             * Creates a new Split for a disabled Stopwatch - <b>called internally only</b>.
063             * Disabled split
064             *
065             * @param stopwatch owning Stopwatch (disabled)
066             */
067            Split(Stopwatch stopwatch) {
068                    assert !(stopwatch.isEnabled()) : "stopwatch must be disabled in this constructor!";
069                    this.enabled = false;
070                    this.stopwatch = stopwatch;
071            }
072    
073            /**
074             * Creates a new Split for direct use without {@link Stopwatch} ("anonymous split"). Stop will not update any Stopwatch,
075             * value can be added to any chosen Stopwatch using {@link Stopwatch#addSplit(Split)} - even in conjunction with
076             * {@link #stop()} like this:
077             * <p/>
078             * <pre>Split split = Split.start();
079             * ...
080             * SimonManager.getStopwatch("codeBlock2.success").addTime(split.stop());</pre>
081             * <p/>
082             * If the split is not needed afterwards calling {@link #stop()} is not necessary:
083             * <p/>
084             * <pre>Split split = Split.start();
085             * ...
086             * SimonManager.getStopwatch("codeBlock2.success").addTime(split);</pre>
087             *
088             * @since 3.4
089             */
090            public static Split start() {
091                    Split split = new Split(true);
092                    split.running = true;
093                    split.start = System.nanoTime();
094                    return split;
095            }
096    
097            /**
098             * Returns the stopwatch that this split is running for. May be {@code null} for anonymous splits (directly created).
099             *
100             * @return owning stopwatch, may be {@code null}
101             */
102            public Stopwatch getStopwatch() {
103                    return stopwatch;
104            }
105    
106            /**
107             * Stops the split, updates the stopwatch and returns this. Subsequent calls do not change the state of the split.
108             *
109             * @return this split object
110             * @since 3.1 - previously returned split time in ns, call additional {@link #runningFor()} for the same result
111             */
112            public Split stop() {
113                    return stop(null);
114            }
115    
116            /**
117             * Stops the split, updates the sub-stopwatch specified by parameter and returns this. Active counter of the original stopwatch
118             * is decreased as expected. Subsequent calls do not change the state of the split.
119             * <p/>
120             * <b>Important:</b> While {@link #stop()} (or this stop with {@code null} argument)
121             * results in {@link org.javasimon.callback.Callback#onStopwatchStop(Split, StopwatchSample)} callback method being invoked,
122             * if sub-simon is affected then {@link org.javasimon.callback.Callback#onStopwatchAdd(Stopwatch, Split, StopwatchSample)}
123             * is called instead.
124             * <p/>
125             * If the split was obtained from disabled Stopwatch, this method does not update sub-simon even if it is enabled, because
126             * split itself is disabled as well. If split is enabled, but sub-simon is disabled, the latter is not updated.
127             *
128             * @param subSimon name of the sub-stopwatch (hierarchy delimiter is added automatically) - if {@code null}
129             * it behaves exactly like {@link #stop()}
130             * @return this split object
131             * @since 3.4
132             */
133            public Split stop(String subSimon) {
134                    if (!running) {
135                            return this;
136                    }
137                    running = false;
138                    long nowNanos = System.nanoTime();
139                    total = nowNanos - start; // we update total before calling the stop so that callbacks can use it
140                    if (stopwatch != null) {
141                            ((StopwatchImpl) stopwatch).stop(this, start, nowNanos, subSimon);
142                    }
143                    return this;
144            }
145    
146            /**
147             * Returns the current running nano-time from the start to the method call or the total split time
148             * if the Split has been stopped already.
149             *
150             * @return current running nano-time of the split
151             */
152            public long runningFor() {
153                    if (!running) {
154                            return total;
155                    }
156                    if (enabled) {
157                            return System.nanoTime() - start;
158                    }
159                    return 0;
160            }
161    
162            /**
163             * Returns printable form of how long this split was running for.
164             *
165             * @return short information about the Split  time as a human readable string
166             * @since 2.2
167             */
168            public String presentRunningFor() {
169                    return SimonUtils.presentNanoTime(runningFor());
170            }
171    
172            /**
173             * Returns true if this split was created from enabled Simon or via {@link #start()}.
174             *
175             * @return true if this split was created from enabled Simon
176             */
177            public boolean isEnabled() {
178                    return enabled;
179            }
180    
181            /**
182             * Returns true if this split is still running ({@link #stop()} has not been called yet).
183             * Returns false for disabled Split.
184             *
185             * @return true if this split is still running
186             * @since 3.1.0
187             */
188            public boolean isRunning() {
189                    return running;
190            }
191    
192            /**
193             * Returns start nano timer value - can be converted to ms timestamp using {@link org.javasimon.utils.SimonUtils#millisForNano(long)}.
194             * Returns 0 if the split is not enabled (started for disabled Stopwatch).
195             *
196             * @return nano timer value when the Split was started or 0 if the split is not enabled
197             * @since 3.1
198             */
199            public long getStart() {
200                    return start;
201            }
202    
203            /**
204             * Stores an attribute in this Split. Attributes can be used to store any custom objects.
205             *
206             * @param name a String specifying the name of the attribute
207             * @param value the Object to be stored
208             * @since 2.3
209             */
210            @Override
211            public void setAttribute(String name, Object value) {
212                    attributesSupport.setAttribute(name, value);
213            }
214    
215            /**
216             * Returns the value of the named attribute as an Object, or {@code null} if no attribute of
217             * the given name exists.
218             *
219             * @param name a String specifying the name of the attribute
220             * @return an Object containing the value of the attribute, or {@code null} if the attribute does not exist
221             * @since 2.3
222             */
223            @Override
224            public Object getAttribute(String name) {
225                    return attributesSupport.getAttribute(name);
226            }
227    
228            /**
229             * Returns the value of the named attribute typed to the specified class, or {@code null} if no attribute of
230             * the given name exists.
231             *
232             * @param name a String specifying the name of the attribute
233             * @return the value of the attribute typed to T, or {@code null} if the attribute does not exist
234             * @since 3.4
235             */
236            @SuppressWarnings("unchecked")
237            @Override
238            public <T> T getAttribute(String name, Class<T> clazz) {
239                    return attributesSupport.getAttribute(name, clazz);
240            }
241    
242            /**
243             * Removes an attribute from this Split.
244             *
245             * @param name a String specifying the name of the attribute to remove
246             * @since 2.3
247             */
248            @Override
249            public void removeAttribute(String name) {
250                    attributesSupport.removeAttribute(name);
251            }
252    
253            /**
254             * Returns an Iterator containing the names of the attributes available to this Split.
255             * This method returns an empty Iterator if the Split has no attributes available to it.
256             *
257             * @return an Iterator of strings containing the names of the Split's attributes
258             * @since 2.3
259             */
260            @Override
261            public Iterator<String> getAttributeNames() {
262                    return attributesSupport.getAttributeNames();
263            }
264    
265            /**
266             * {@inheritDoc}
267             *
268             * @since 3.4
269             */
270            @Override
271            public Map<String, Object> getCopyAsSortedMap() {
272                    return attributesSupport.getCopyAsSortedMap();
273            }
274    
275            /**
276             * Returns information about this Split, if it's running, name of the related Stopwatch and split's time.
277             *
278             * @return information about the Split as a human readable string
279             */
280            @Override
281            public String toString() {
282                    if (!enabled) {
283                            return "Split created from disabled Stopwatch";
284                    }
285                    if (running) {
286                            return "Running split" + (stopwatch != null ? " for Stopwatch '" + stopwatch.getName() + "': " : ": ") + SimonUtils.presentNanoTime(runningFor());
287                    }
288                    return "Stopped split" + (stopwatch != null ? " for Stopwatch '" + stopwatch.getName() + "': " : ": ") + SimonUtils.presentNanoTime(total);
289            }
290    }
