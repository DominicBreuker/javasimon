001    package org.javasimon.console.action;
002    
003    import org.javasimon.console.text.BaseStringifier;
004    import org.javasimon.console.text.Stringifier;
005    import org.javasimon.console.text.StringifierFactory;
006    
007    /**
008     * Value formatter for CSV reponse format
009     * @author gquintana
010     */
011    public class CsvStringifierFactory extends StringifierFactory {
012            @Override
013            protected Stringifier<String> registerStringStringifier(Stringifier nullStringifier) {
014                    Stringifier<String> stringStringifier=new BaseStringifier<String>(nullStringifier) {
015                            @Override
016                            protected String doToString(String s) {
017                                    return"\"" + s.replace("\"", "\"\"") + "\"";
018                            }
019                    };
020                    compositeStringifier.add(String.class, stringStringifier);
021                    return stringStringifier;
022            }
023    };
