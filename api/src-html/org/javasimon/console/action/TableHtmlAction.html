001    package org.javasimon.console.action;
002    
003    import org.javasimon.console.html.HtmlBuilder;
004    import java.io.IOException;
005    import java.io.PrintWriter;
006    import java.util.Collections;
007    import org.javasimon.Sample;
008    import org.javasimon.console.ActionContext;
009    import org.javasimon.console.SimonType;
010    import org.javasimon.console.text.StringifierFactory;
011    
012    /**
013     * Export Simons as a flat static HTML table which can be used for printing.
014     * Each row as an odd/even CSS class.
015     * Each cell as a CSS class corresponding to property name
016     * Only a subset (definined in {@link AbstractTableAction#columns}) of attributes are exported.
017     * Path: http://.../data/table.html?pattern=SimonPattern&type=STOPWATCH&type=COUNTER&timeFormat=MILLISECOND
018     *
019     * @author gquintana
020     */
021    public class TableHtmlAction extends AbstractTableAction {
022            /**
023             * Path to display Flat HTML table
024             */
025            public static final String PATH = "/data/table.html";
026            /**
027             * Constructor.
028             */
029            public TableHtmlAction(ActionContext context) {
030                    super(context, "text/html");
031                    this.stringifierFactory = new StringifierFactory();
032                    this.numberPattern=StringifierFactory.INTEGER_NUMBER_PATTERN;
033            }
034            /**
035             * Row index
036             */
037            private Integer index;
038            /**
039             * HTML Builder
040             */
041            private HtmlBuilder htmlBuilder;
042            /**
043             * {@inheritDoc }
044             */
045            @Override
046            protected void printTable(PrintWriter writer) throws IOException {
047                    htmlBuilder=new HtmlBuilder(writer);
048                    htmlBuilder.header("List View", Collections.emptyList())
049                            .begin("table", "flatTable", "flatTable");
050                    super.printTable(writer);
051                    htmlBuilder.end("table").footer();
052            }
053    
054            /**
055             * {@inheritDoc }
056             */
057            @Override
058            protected void printHeaderRow(PrintWriter writer) throws IOException {
059                    htmlBuilder.begin("thead").begin("tr");
060                    super.printHeaderRow(writer);
061                    htmlBuilder.end("tr").end("thead");
062            }
063    
064            /**
065             * {@inheritDoc }
066             */
067            @Override
068            protected void printBody(PrintWriter writer) throws IOException {
069                    htmlBuilder.begin("tbody");
070                    index = 0;
071                    super.printBody(writer);
072                    index = null;
073                    htmlBuilder.end("tbody");
074            }
075    
076            /**
077             * {@inheritDoc }
078             */
079            @Override
080            protected void printBodyRow(Sample sample, PrintWriter writer) throws IOException {
081                    htmlBuilder.begin("tr",Integer.toString(index), index % 2 == 0 ? "even" : "odd");
082                    super.printBodyRow(sample, writer);
083                    index++;
084                    htmlBuilder.end("tr");
085            }
086    
087            /**
088             * {@inheritDoc }
089             */
090            @Override
091            protected void printCell(Column column, String s, PrintWriter writer) throws IOException {
092                    htmlBuilder.begin("td",null, column.getName());
093                    if (column.getName().equals("type")) {
094                            try {
095                                    htmlBuilder.simonTypeImg(SimonType.valueOf(s), "../../");
096                            } catch(IllegalArgumentException illegalArgumentException) {
097                                    // Else unknown type: non image
098                            }
099                    }
100                    super.printCell(column, s, writer);
101                    htmlBuilder.end("td");
102            }
103    }
