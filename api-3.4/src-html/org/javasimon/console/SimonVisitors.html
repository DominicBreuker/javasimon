001    package org.javasimon.console;
002    
003    import java.io.IOException;
004    import java.util.*;
005    
006    import org.javasimon.Manager;
007    import org.javasimon.Simon;
008    import org.javasimon.SimonPattern;
009    
010    /**
011     * Helper class to navigate through Simons.
012     *
013     * @author gquintana
014     */
015    public class SimonVisitors {
016    
017            private SimonVisitors() {
018            }
019    
020            /**
021             * Visit simons as a list.
022             * Calls {@link  Manager#getSimons(org.javasimon.SimonPattern)} method
023             * then Simons are sorted by name and filtered by type
024             * finally the visitor is called on each of them.
025             *
026             * @param manager Simon manager
027             * @param pattern Pattern
028             * @param type Type
029             * @param visitor Visitor
030             * @throws IOException
031             */
032            public static void visitList(Manager manager, String pattern, Set<SimonType> types, SimonVisitor visitor) throws IOException {
033                    List<Simon> simons = new ArrayList<Simon>(manager.getSimons(SimonPattern.create(pattern)));
034                    Collections.sort(simons, new Comparator<Simon>() {
035                            public int compare(Simon s1, Simon s2) {
036                                    return s1.getName().compareTo(s2.getName());
037                            }
038                    });
039                    for (Simon simon : simons) {
040                            SimonType lType = SimonTypeFactory.getValueFromInstance(simon);
041                            if (types == null || types.contains(lType)) {
042                                    visitor.visit(simon);
043                            }
044                    }
045            }
046    
047            /**
048             * Visit Simons recursively as a tree starting from the specified name or root Simon if the name is {@code null}.
049             *
050             * @param manager Simon manager
051             * @param visitor Visitor
052             * @throws IOException
053             */
054            public static void visitTree(Manager manager, String rootName, SimonVisitor visitor) throws IOException {
055                    Simon root = rootName == null ? manager.getRootSimon() : manager.getSimon(rootName);
056                    visitTree(root, visitor);
057            }
058    
059            /**
060             * Visit Simons recursively as a tree starting from the specified Simon.
061             *
062             * @param simon Parent simon
063             * @param visitor Visitor
064             * @throws IOException
065             */
066            public static void visitTree(Simon simon, SimonVisitor visitor) throws IOException {
067                    visitor.visit(simon);
068                    for (Simon childSimon : simon.getChildren()) {
069                            visitTree(childSimon, visitor);
070                    }
071            }
072    }
