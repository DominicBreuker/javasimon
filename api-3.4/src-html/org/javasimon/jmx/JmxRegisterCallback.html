001    package org.javasimon.jmx;
002    
003    import org.javasimon.*;
004    import org.javasimon.callback.CallbackSkeleton;
005    
006    import javax.management.MBeanServer;
007    import javax.management.ObjectName;
008    import javax.management.JMException;
009    import java.util.Set;
010    import java.util.Iterator;
011    import java.util.HashSet;
012    import java.lang.management.ManagementFactory;
013    
014    /**
015     * Callback that registers MXBeans for Simons after their creation. It is
016     * advisable to register the callback as soon as possible otherwise MX Beans
017     * for some Simons may not be created. Class can be extended in order to
018     * override {@link #constructObjectName(Simon)}.
019     *
020     * @author <a href="mailto:virgo47@gmail.com">Richard "Virgo" Richter</a>
021     */
022    public class JmxRegisterCallback extends CallbackSkeleton {
023            /**
024             * Domain part of the JMX object name - protected for subclasses.
025             */
026            protected String domain;
027    
028            /**
029             * MBean server instance specified for this callback (or default platform one) - protected for subclasses.
030             */
031            protected MBeanServer mBeanServer;
032    
033            private Set<String> registeredNames = new HashSet<String>();
034    
035            /**
036             * Default constructor uses default MBeanServer.
037             *
038             * @param domain domain part of the object name
039             */
040            public JmxRegisterCallback(String domain) {
041                    this(ManagementFactory.getPlatformMBeanServer(), domain);
042            }
043    
044            /**
045             * Constructor using specific MBeanServer.
046             *
047             * @param mBeanServer specific MBeanServer
048             * @param domain domain part of the object name
049             */
050            public JmxRegisterCallback(MBeanServer mBeanServer, String domain) {
051                    assert domain != null && !(domain.isEmpty());
052                    this.mBeanServer = mBeanServer;
053                    this.domain = domain;
054            }
055    
056            /**
057             * After Simon is created respective MX bean is registered for it according to
058             * its type.
059             *
060             * @param simon created Simon
061             */
062            @Override
063            public final void onSimonCreated(Simon simon) {
064                    if (simon.getName() == null) {
065                            return;
066                    }
067                    register(simon);
068            }
069    
070            /**
071             * When the Simon is destroyed, its MX bean is unregistered.
072             *
073             * @param simon destroyed Simon
074             */
075            @Override
076            public final void onSimonDestroyed(Simon simon) {
077                    String name = constructObjectName(simon);
078                    try {
079                            ObjectName objectName = new ObjectName(name);
080                            mBeanServer.unregisterMBean(objectName);
081                            registeredNames.remove(name);
082                            onManagerMessage("Unregistered Simon with the name: " + objectName);
083                    } catch (JMException e) {
084                            onManagerWarning("JMX unregistration failed for: " + name, e);
085                    }
086            }
087    
088            /**
089             * When the manager is cleared, all MX beans for its Simons are unregistered.
090             */
091            @Override
092            public final void onManagerClear() {
093                    Iterator<String> namesIter = registeredNames.iterator();
094                    while (namesIter.hasNext()) {
095                            String name = namesIter.next();
096                            try {
097                                    ObjectName objectName = new ObjectName(name);
098                                    mBeanServer.unregisterMBean(objectName);
099                                    // I have to use iterator.remove() - hence no common method for clearManager() and simonDestroyed(simon)
100                                    namesIter.remove();
101                                    onManagerMessage("Unregistered Simon with the name: " + objectName);
102                            } catch (JMException e) {
103                                    onManagerWarning("JMX unregistration failed for: " + name, e);
104                            }
105                    }
106            }
107    
108            /**
109             * Method registering Simon MX Bean - can not be overridden, but can be used in subclasses.
110             *
111             * @param simon Simon MX Bean to be registered
112             * @since 3.1
113             */
114            protected final void register(Simon simon) {
115                    Object mBean = constructObject(simon);
116                    String name = constructObjectName(simon);
117                    if (mBean != null && name != null) {
118                            try {
119                                    ObjectName objectName = new ObjectName(name);
120                                    if (mBeanServer.isRegistered(objectName)) {
121                                            mBeanServer.unregisterMBean(objectName);
122                                    } else {
123                                            registeredNames.add(name);
124                                    }
125                                    mBeanServer.registerMBean(mBean, objectName);
126                                    onManagerMessage("Simon registered under the name: " + objectName);
127                            } catch (JMException e) {
128                                    onManagerWarning("JMX registration failed for: " + name, e);
129                                    registeredNames.remove(name);
130                            }
131                    }
132            }
133    
134            /**
135             * Constructs JMX object from Simon object. Method can be overridden.
136             *
137             * @param simon Simon object
138             * @return JMX object (=MBean) representing the Simon
139             */
140            protected SimonSuperMXBean constructObject(Simon simon) {
141                    SimonSuperMXBean simonMxBean;
142                    if (simon instanceof Counter) {
143                            simonMxBean = new CounterMXBeanImpl((Counter) simon);
144                    } else if (simon instanceof Stopwatch) {
145                            simonMxBean = new StopwatchMXBeanImpl((Stopwatch) simon);
146                    } else {
147                            onManagerWarning("Unknown type of Simon! " + simon, null);
148                            simonMxBean = null;
149                    }
150                    return simonMxBean;
151            }
152    
153            /**
154             * Constructs JMX object name from Simon object. Method can be overridden.
155             *
156             * @param simon Simon object
157             * @return object name in String form
158             */
159            protected String constructObjectName(Simon simon) {
160                    return domain + ":type=" + simonType(simon) + ",name=" + simon.getName();
161            }
162    
163            /**
164             * Returns type of the simon as defined in {@link SimonInfo#COUNTER},
165             * {@link SimonInfo#STOPWATCH} or {@link SimonInfo#UNKNOWN}.
166             *
167             * @param simon Simon object
168             * @return type of the Simon as String
169             */
170            protected String simonType(Simon simon) {
171                    String type = SimonInfo.UNKNOWN;
172                    if (simon instanceof Counter) {
173                            type = SimonInfo.COUNTER;
174                    } else if (simon instanceof Stopwatch) {
175                            type = SimonInfo.STOPWATCH;
176                    }
177                    return type;
178            }
179    }
