001    package org.javasimon.console.action;
002    
003    import java.io.IOException;
004    import java.io.InputStream;
005    import java.io.OutputStream;
006    import java.util.HashMap;
007    import java.util.Map;
008    import javax.servlet.ServletException;
009    import org.javasimon.console.Action;
010    import org.javasimon.console.ActionContext;
011    import org.javasimon.console.ActionException;
012    
013    /**
014     * Action used to send resources (images, JS, CSS) to the browser
015     * @author gquintana
016     */
017    public class ResourceAction extends Action {
018    
019            public static final String PREFIX = "/resource";
020            private static final Map<String, String> CONTENT_TYPES = new HashMap<String, String>();
021            /* Fill the CONTENT_TYPES map
022             */
023            static {
024                    CONTENT_TYPES.put("gif", "image/gif");
025                    CONTENT_TYPES.put("png", "image/png");
026                    CONTENT_TYPES.put("jpg", "image/jpeg");
027                    CONTENT_TYPES.put("html", "text/html");
028                    CONTENT_TYPES.put("css", "text/css");
029                    CONTENT_TYPES.put("js", "text/javascript");
030            }
031            /**
032             * Relative path of the resource to return
033             */
034            private final String resourcePath;
035            /**
036             * Constructor
037             * @param context Action context
038             * @param resourcePath Local path of the resource
039             */
040            public ResourceAction(ActionContext context, String resourcePath) {
041                    super(context);
042                    if (resourcePath.startsWith("/")) {
043                            this.resourcePath=resourcePath;
044                    } else {
045                            this.resourcePath="/"+resourcePath;
046                    }
047            }
048    
049            @Override
050            public void execute() throws ServletException, IOException, ActionException {
051                    InputStream resourceIStream = null;
052                    try {
053                            resourceIStream = getClass().getResourceAsStream("/org/javasimon/console/resource" + resourcePath);
054                            if (resourceIStream == null) {
055                                    throw new ActionException("Resource " + resourcePath + " not found");
056                            }
057                            String extension = resourcePath.substring(resourcePath.lastIndexOf('.') + 1).toLowerCase();
058                            String contentType = CONTENT_TYPES.get(extension);
059                            if (contentType != null) {
060                                    getContext().setContentType(contentType);
061                                    if (contentType.startsWith("text")) {
062                                            getContext().getResponse().setCharacterEncoding("UTF-8");
063                                    }
064                            }
065                            getContext().getResponse().setHeader("Cache-Control", "public; max-age=300");
066                            copyStream(resourceIStream);
067                    } finally {
068                            if (resourceIStream != null) {
069                                    resourceIStream.close();
070                            }
071                    }
072            }
073            /**
074             * Copy resource input stream to HTTP response output stream using
075             * a 64kib buffer
076             */
077            private void copyStream(InputStream inputStream) throws IOException {
078                    OutputStream outputStream = null;
079                    try {
080                            outputStream = getContext().getOutputStream();
081                            byte[] buffer = new byte[65535];
082                            int bufferLen;
083                            int totalLen = 0;
084                            while ((bufferLen = inputStream.read(buffer)) > 0) {
085                                    outputStream.write(buffer, 0, bufferLen);
086                                    totalLen += bufferLen;
087                            }
088                            getContext().getResponse().setContentLength(totalLen);
089                    } finally {
090                            if (outputStream != null) {
091                                    outputStream.flush();
092                            }
093                    }
094            }
095    }
