001    package org.javasimon.callback.timeline;
002    
003    import org.javasimon.Split;
004    import org.javasimon.utils.SimonUtils;
005    
006    /**
007     * Stopwatch statistics gathered for a specific time range.
008     * This class is not intended to be thread safe.
009     *
010     * @author gerald
011     */
012    public class StopwatchTimeRange extends TimeRange {
013            /**
014             * Last value
015             */
016            private long last;
017            /**
018             * Minimum
019             */
020            private long min = Long.MAX_VALUE;
021            /**
022             * Minimum
023             */
024            private long max = Long.MIN_VALUE;
025            /**
026             * Total (Sum of all values)
027             */
028            private long total;
029            /**
030             * (Sum of square of all values)
031             */
032            private long squareTotal;
033            /**
034             * Counter (Number of values)
035             */
036            private long counter;
037    
038            /**
039             * Main constructor.
040             */
041            public StopwatchTimeRange(long startTimestamp, long endTimestamp) {
042                    super(startTimestamp, endTimestamp);
043            }
044    
045            /**
046             * Add stopwatch split information.
047             * @param timestampInMs When the split started, expressed in milliseconds
048             * @param durationInNs How long the split was, expressed in nanoseconds
049             */
050            public void addSplit(long timestampInMs, long durationInNs) {
051                    last = durationInNs;
052                    total += durationInNs;
053                    squareTotal += durationInNs * durationInNs;
054                    if (durationInNs > max) {
055                            max = durationInNs;
056                    }
057                    if (durationInNs < min) {
058                            min = durationInNs;
059                    }
060                    counter++;
061                    lastTimestamp = timestampInMs;
062            }
063    
064            public long getLast() {
065                    return last;
066            }
067    
068            public long getMin() {
069                    return min;
070            }
071    
072            public long getMax() {
073                    return max;
074            }
075    
076            public long getTotal() {
077                    return total;
078            }
079    
080            public long getCounter() {
081                    return counter;
082            }
083            /**
084             * Compute mean/average using total and counter
085             */
086            private double computeMean() {
087                    return ((double) total) / ((double) counter);
088            }
089    
090            /**
091             * Compute mean/average.
092             * @return mean/average duration.
093             */
094            public Double getMean() {
095                    return counter == 0 ? null : computeMean();
096            }
097    
098            /**
099             * Compute variance.
100             * @return variance
101             */
102            public Double getVariance() {
103                    if (counter == 0) {
104                            return null;
105                    } else {
106                            final double mean = computeMean();
107                            final double meanSquare = mean * mean;
108                            final double squareMean = ((double) squareTotal) / ((double) counter);
109                            return squareMean - meanSquare;
110                    }
111            }
112            /**
113             * Compute standard deviation.
114             * @return Standard deviation
115             */
116            public Double getStandardDeviation() {
117                    final Double variance = getVariance();
118                    return variance == null ? null : Math.sqrt(variance);
119            }
120    
121            @Override
122            protected StringBuilder toStringBuilder(StringBuilder stringBuilder) {
123                    return super.toStringBuilder(stringBuilder)
124                            .append(" counter=").append(counter)
125                            .append(" total=").append(SimonUtils.presentNanoTime(total))
126                            .append(" min=").append(SimonUtils.presentNanoTime(min))
127                            .append(" mean=").append(SimonUtils.presentNanoTime(getMean()))
128                            .append(" last=").append(SimonUtils.presentNanoTime(getLast()))
129                            .append(" max=").append(SimonUtils.presentNanoTime(max))
130                            .append(" stddev=").append(SimonUtils.presentNanoTime(getStandardDeviation()));
131            }
132    
133    }
