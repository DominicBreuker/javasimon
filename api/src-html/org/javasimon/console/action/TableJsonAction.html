001    package org.javasimon.console.action;
002    
003    import java.io.IOException;
004    import java.io.PrintWriter;
005    import org.javasimon.Sample;
006    import org.javasimon.console.ActionContext;
007    import org.javasimon.console.json.ArrayJS;
008    import org.javasimon.console.json.JsonStringifierFactory;
009    import org.javasimon.console.json.ObjectJS;
010    import org.javasimon.console.json.SimpleJS;
011    
012    /**
013     * Export Simons as a flat JSON array to be displayed in datatable.
014     * Each JSONized simon as all attributes even if not present on real Simon.
015     * Only a subset (definined in {@link AbstractTableAction#columns}) of attributes are exported.
016     *
017     * @author gquintana
018     */
019    public class TableJsonAction extends AbstractTableAction {
020    
021            public static final String PATH = "/data/table.json";
022    
023            public TableJsonAction(ActionContext context) {
024                    super(context, "application/json");
025                    this.stringifierFactory = new JsonStringifierFactory();
026                    this.numberPattern=JsonStringifierFactory.INTEGER_NUMBER_PATTERN;
027            }
028            /**
029             * Current array of JSONized Simons 
030             */
031            private ArrayJS simonsJS;
032            /**
033             * Current JSONized Simon
034             */
035            private ObjectJS simonJS;
036    
037            @Override
038            protected void printTable(PrintWriter writer) throws IOException {
039                    simonsJS = new ArrayJS();
040                    super.printTable(writer);
041                    simonsJS.write(writer);
042                    simonJS = null;
043            }
044    
045            @Override
046            protected void printHeaderRow(PrintWriter writer) throws IOException {
047                    // No header
048            }
049    
050            @Override
051            protected void printBodyRow(Sample sample, PrintWriter writer) throws IOException {
052                    simonJS = new ObjectJS();
053                    simonsJS.addElement(simonJS);
054                    super.printBodyRow(sample, writer);
055                    simonJS = null;
056            }
057    
058            @Override @SuppressWarnings("unchecked")
059            protected void printBodyCell(Column column, Sample sample, PrintWriter writer) {
060                    simonJS.setAttribute(column.getName(), new SimpleJS(column.getValue(sample), column.getStringifier(sample)));
061            }
062    }
