001    package org.javasimon;
002    
003    import java.util.List;
004    
005    /**
006     * Simon interface contains common functions related to Simon management - enable/disable and hierarchy.
007     * It does not contain any real action method - these are in specific interfaces that describes
008     * purpose of the particular type of monitor.
009     *
010     * @author <a href="mailto:virgo47@gmail.com">Richard "Virgo" Richter</a>
011     * @see Manager
012     * @see Counter for Simon counting some events
013     * @see Stopwatch for Simon measuring time spans
014     */
015    public interface Simon extends HasAttributes {
016            /**
017             * Returns Simon name. Simon name is always fully qualified
018             * and determines also position of the Simon in the monitor hierarchy.
019             * Simon name can be {@code null} for anonymous Simons.
020             *
021             * @return name of the Simon
022             */
023            String getName();
024    
025            /**
026             * Returns parent Simon.
027             *
028             * @return parent Simon
029             */
030            Simon getParent();
031    
032            /**
033             * Returns list of children - direct sub-simons.
034             *
035             * @return list of children
036             */
037            List<Simon> getChildren();
038    
039            /**
040             * Returns state of the Simon that can be enabled, disabled or ihnerited.
041             *
042             * @return state of the Simon
043             */
044            SimonState getState();
045    
046            /**
047             * Sets the state of the Simon. It must be specified whether to propagate the change
048             * and overrule states of all sub-simons which effectively sets the same state to the whole
049             * subtree. If subtree is not overruled, some Simons (with their subtrees) may not be affected
050             * by this change.
051             *
052             * @param state a new state.
053             * @param overrule specifies whether this change is forced to the whole subtree.
054             */
055            void setState(SimonState state, boolean overrule);
056    
057            /**
058             * Returns true, if the Simon is enabled or if the enabled state is inherited.
059             *
060             * @return true, if the Simon is effectively enabled
061             */
062            boolean isEnabled();
063    
064            /**
065             * Resets the Simon values related to the measuring, timestamps and so on - usage timestamps, state, attributes are not affected.
066             * Timestamp of the last reset can be obtained by the method {@link #getLastReset()}. Reset is perfomed even for disabled Simons.
067             *
068             * @return returns this
069             */
070            Simon reset();
071    
072            /**
073             * Returns ms timestamp of the last recent usage of the {@link #reset()} method on the Simon.
074             * Returns 0 if {@code reset} was not called yet. This timestamp is useful for rate measuring
075             * when reset is called on a regular basis - likely via {@link #sampleAndReset()}. While
076             * client code could store the timestamp too it is not necessary with this method.
077             *
078             * @return ms timestamp of the last reset or 0 if reset was not called yet
079             */
080            long getLastReset();
081    
082            /**
083             * Returns note for the Simon. Note enables Simon with an additional information in human
084             * readable form.
085             *
086             * @return note for the Simon.
087             */
088            String getNote();
089    
090            /**
091             * Sets note for the Simon. Note enables Simon with an additional information in human
092             * readable form.
093             *
094             * @param note note for the Simon.
095             */
096            void setNote(String note);
097    
098            /**
099             * Returns ms timestamp of the first usage of this Simon. First and last usage
100             * are updated when monitor performs the measuring (start/stop/count/etc). They
101             * are not updated when values are obtained from the monitor.
102             *
103             * @return ms timestamp of the first usage
104             */
105            long getFirstUsage();
106    
107            /**
108             * Returns ms timestamp of the last usage of this Simon. First and last usage
109             * are updated when monitor performs the measuring (start/stop/count/etc). They
110             * are not updated when values are obtained from the monitor.
111             *
112             * @return ms timestamp of the last usage
113             */
114            long getLastUsage();
115    
116            /**
117             * Samples Simon values and returns them in a Java Bean derived from Sample interface.
118             *
119             * @return sample containing all Simon values
120             */
121            Sample sample();
122    
123            /**
124             * Samples Simon values and returns them in a Java Bean derived from Sample interface
125             * and resets the Simon. Operation is synchronized to assure atomicity.
126             *
127             * @return sample containing all Simon values
128             */
129            Sample sampleAndReset();
130    }
